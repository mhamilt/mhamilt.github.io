    <script>

    class Tone {
        constructor(osc, env) {
            this.osc = osc;
            this.env = env;
        }
        play() {
            this.osc.start()
            this.env.play(this.osc);
        }

        setFreq(freq) {
            this.osc.freq(freq)
        }
    }
    //------------------------------
    // FUNCTIONS
    //------------------------------
    function midi2Hz(midiNoteNumber) {
        return Math.pow(2, (midiNoteNumber - 69) / 12.0) * 415.0;
    }

    //------------------------------
    // CONSTANTS
    //------------------------------

    const tones = new Array(2);
    const meantone = [
        310.51,
        332.14,
        346.86,
        371.50,
        388.03,
        415.26,
        444.24,
        464.18,
        496.60,
        518.70,
        555.29,
        593.70,
    ]

    window.preload = function () {
        const attack = 0.001;
        const decay = 0.05;
        const sustain = 0.707;
        const release = 1.75;
        const gain = .707;

        for (let i = 0; i < 2; i++) {
            tones[i] = new Tone(new p5.Oscillator(), new p5.Envelope(attack, gain, decay, gain));
            tones[i].env.setADSR(attack, decay, sustain, release);
            tones[i].env.setRange(gain, 0);
            tones[i].env.setExp(false);
        }

        console.log("preload");
    };

    window.setup = function () {
        createCanvas(1, 1); // can even be minimal
    };

    const table = document.getElementById("temperament-table");
    const cells = table.querySelectorAll("td");

    cells.forEach(cell => {
        const row = cell.parentElement.rowIndex;
        const col = cell.cellIndex;

        if (row === 0 || col === 0) {
            // cell.style.fontSize = '2em'
        }

        cell.addEventListener("mouseenter", () => {
            clearHighlights();


            // Highlight row up to hovered cell
            for (let c = 0; c <= col; c++) {
                table.rows[row].cells[c].classList.add("active");
            }

            // Highlight column up to hovered cell
            for (let r = 0; r <= row; r++) {
                table.rows[r].cells[col].classList.add("active");
            }

            // Highlight current cell separately
            cell.classList.add("current");
        });

        if (row > 0 && col > 0) {
            cell.addEventListener("mousedown", () => {
                if (row === col) {
                    tones[0].setFreq(midi2Hz(row + 63));
                    tones[1].setFreq(midi2Hz(col + 63 + 12));
                }
                else if (row > col) {
                    tones[0].setFreq(midi2Hz(row + 63));
                    tones[1].setFreq(midi2Hz(col + 63));
                }
                else if (col > row) {
                    tones[0].setFreq(meantone[col - 1])
                    tones[1].setFreq(meantone[row - 1])
                }
                tones[0].play();
                tones[1].play();                    
            });
        }
    });

    table.addEventListener("mouseleave", clearHighlights);

    function clearHighlights() {
        cells.forEach(cell => {
            cell.classList.remove("active", "current");
        });
    }
</script>
